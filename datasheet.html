在操作高压外设前，请确保 VDD 处于 3.3V 稳定状态。

3.1. UART 初始化

以下是基于 C 语言 的初始化代码示例：

#include "mcu_uart.h"

void uart_init(uint32_t baudrate) {
    UART->BAUD = SystemCoreClock / baudrate; // 设置波特率
    UART->CTRL |= UART_ENABLE;               // 使能串口
}

4.1.3. C Example (Read-Modify-Write)

#include <stdint.h>

#define REG32(addr) (*(volatile uint32_t *)(addr))

#define CTRL_BASE   0x40020000UL
#define CTRL_OFF    0x00UL
#define CTRL        REG32(CTRL_BASE + CTRL_OFF)

#define CTRL_SPEED_Pos   16u
#define CTRL_SPEED_Msk   (0xFFu << CTRL_SPEED_Pos)

#define CTRL_PULL_Pos    12u
#define CTRL_PULL_Msk    (0xFu  << CTRL_PULL_Pos)

#define CTRL_MODE_Pos    8u
#define CTRL_MODE_Msk    (0xFu  << CTRL_MODE_Pos)

#define CTRL_EN_Pos      7u
#define CTRL_EN_Msk      (1u    << CTRL_EN_Pos)

#define CTRL_INTEN_Pos   6u
#define CTRL_INTEN_Msk   (1u    << CTRL_INTEN_Pos)

#define CTRL_KEY_Pos     0u
#define CTRL_KEY_Msk     (0x3Fu << CTRL_KEY_Pos)
#define CTRL_KEY_VALUE   0x2Au

static inline void ctrl_config(uint32_t mode, uint32_t pull, uint32_t speed, uint32_t en)
{
    uint32_t reg = CTRL;

    // Preserve reserved bits; only modify documented fields.
    reg &= ~(CTRL_MODE_Msk | CTRL_PULL_Msk | CTRL_SPEED_Msk | CTRL_EN_Msk | CTRL_KEY_Msk);
    reg |= ((mode  & 0xFu)  << CTRL_MODE_Pos);
    reg |= ((pull  & 0xFu)  << CTRL_PULL_Pos);
    reg |= ((speed & 0xFFu) << CTRL_SPEED_Pos);
    reg |= ((en ? 1u : 0u)  << CTRL_EN_Pos);
    reg |= (CTRL_KEY_VALUE  << CTRL_KEY_Pos);

    CTRL = reg;
}

6. 寄存器描述 32-bit Register Example

6.1. GPIO_MODE Register (0x00)

6.1.1. Register Layout (32-bit)

Bits Field Reset Description

31:16

RESERVED

0x0000

Reserved, must keep default value.

15:8

SPEED

0x00

Output speed selection.

7:4

PULL

0x0

Pull-up / pull-down configuration.

3:0

MODE

0x0

Pin mode selection.

6.1.2. Field Definitions

  • MODE (Bits [3:0])

    • 0x0 = Input

    • 0x1 = Output

    • 0x2 = Alternate Function

    • 0x3 = Analog

  • PULL (Bits [7:4])

    • 0x0 = No pull

    • 0x1 = Pull-up

    • 0x2 = Pull-down

  • SPEED (Bits [15:8])

    • 0x00 = Low

    • 0x01 = Medium

    • 0x02 = High

    • 0x03 = Very High

6.1.3. C Access Example

#include <stdint.h>

#define GPIO_BASE      (0x40020000UL)
#define GPIO_MODE_OFF  (0x00UL)
#define GPIO_MODE      (*(volatile uint32_t *)(GPIO_BASE + GPIO_MODE_OFF))

// Bit positions / masks
#define GPIO_MODE_MODE_Pos   0
#define GPIO_MODE_MODE_Msk   (0xFUL << GPIO_MODE_MODE_Pos)

#define GPIO_MODE_PULL_Pos   4
#define GPIO_MODE_PULL_Msk   (0xFUL << GPIO_MODE_PULL_Pos)

#define GPIO_MODE_SPEED_Pos  8
#define GPIO_MODE_SPEED_Msk  (0xFFUL << GPIO_MODE_SPEED_Pos)

static inline void gpio_mode_set(uint32_t mode, uint32_t pull, uint32_t speed)
{
    uint32_t reg = GPIO_MODE;

    reg &= ~(GPIO_MODE_MODE_Msk | GPIO_MODE_PULL_Msk | GPIO_MODE_SPEED_Msk);
    reg |= ((mode  & 0xFUL)  << GPIO_MODE_MODE_Pos);
    reg |= ((pull  & 0xFUL)  << GPIO_MODE_PULL_Pos);
    reg |= ((speed & 0xFFUL) << GPIO_MODE_SPEED_Pos);

    GPIO_MODE = reg;
}

static inline uint32_t gpio_mode_get(void)
{
    return GPIO_MODE;
}

Notes

  • This register is 32-bit aligned, so offsets increment by 4 bytes (0x00, 0x04, 0x08…​).

  • Always preserve RESERVED bits (read-modify-write) unless the spec says otherwise.
